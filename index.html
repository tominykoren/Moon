/** * פונקציית GET לבדיקת זמינות השרת * מחזירה אישור טקסטואלי פשוט שהמערכת באוויר. */ function doGet(e) { return ContentService.createTextOutput("השרת עובד בהצלחה!"); } /** * פונקציית העזר להמרת תאריכים לפורמט YYYY-MM-DD * מבטיחה אחידות במפתח התאריכים (dateMap) בין מקורות שונים */ function formatUniformDate(dateStringOrObj) { let d = new Date(dateStringOrObj); if (isNaN(d.getTime())) return null; let month = '' + (d.getMonth() + 1); let day = '' + d.getDate(); let year = d.getFullYear(); if (month.length < 2) month = '0' + month; if (day.length < 2) day = '0' + day; return [year, month, day].join('-'); } /** * פונקציית POST המרכזית * אוספת נתונים מיאהו (או גוגל), מסדרת אותם בתצוגת ציר-זמן רוחבית (Pivot), * ומחזירה כ-CSV או גיליון. */ function doPost(e) { try { // --- 1. פענוח בטוח של המידע (CRITICAL Payload Parsing) --- let params = {}; try { const raw = e && e.postData && e.postData.contents ? e.postData.contents : (e && e.postData && e.postData.getDataAsString ? e.postData.getDataAsString() : "{}"); params = JSON.parse(raw); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Parse failed: " + err.message })).setMimeType(ContentService.MimeType.JSON); } // וידוא חובה של המפתחות המדויקים const { tickers, startDate, endDate, frequency, action } = params; if (!tickers || !startDate || !endDate || !frequency || !action) { throw new Error("Missing parameters. Received: " + JSON.stringify(params)); } // --- 2. הכנת הנתונים והמשתנים --- // פיצול המניות, ניקוי רווחים והמרה לאותיות גדולות const tickersArray = tickers.split(",").map(t => t.trim().toUpperCase()).filter(t => t.length > 0); // אובייקט לאחסון מחירי מניות לפי תאריך (המפתח יהיה התאריך) // מבנה: { "2023-01-01": { "AAPL": 150.5, "MSFT": 250.2 }, ... } const dateMap = {}; // התאמת פרמטר התדירות עבור Yahoo ו-Google const freq = frequency.toLowerCase(); let yahooFreq = "1d"; let gfFreq = "DAILY"; if (freq === "weekly") { yahooFreq = "1wk"; gfFreq = "WEEKLY"; } else if (freq === "monthly") { yahooFreq = "1mo"; gfFreq = "WEEKLY"; // Google Finance לא תומך היסטורית בחודשי } // חותמות זמן ליאהו (תיקון ל-UTC כדי למנוע סטיות תאריכים) const period1 = Math.floor(new Date(startDate + "T00:00:00Z").getTime() / 1000); const period2 = Math.floor(new Date(endDate + "T23:59:59Z").getTime() / 1000); // --- 3. לולאת איסוף הנתונים לכל מניה --- for (let i = 0; i < tickersArray.length; i++) { const ticker = tickersArray[i]; try { // --- ניסיון ראשון: Yahoo Finance --- const yahooUrl = `https://query1.finance.yahoo.com/v7/finance/download/${encodeURIComponent(ticker)}?period1=${period1}&period2=${period2}&interval=${yahooFreq}&events=history&includeAdjustedClose=true`; const response = UrlFetchApp.fetch(yahooUrl, { muteHttpExceptions: true }); if (response.getResponseCode() !== 200) { throw new Error(`Yahoo Finance Error: ${response.getResponseCode()}`); } const csvContent = response.getContentText(); const parsedCsv = Utilities.parseCsv(csvContent); if (parsedCsv.length < 2) throw new Error("No data returned from Yahoo"); // איתור העמודות הרלוונטיות דינמית const header = parsedCsv[0]; let dateIdx = header.indexOf("Date"); let closeIdx = header.indexOf("Adj Close"); if (closeIdx === -1) closeIdx = header.indexOf("Close"); // גיבוי אם אין מחיר מתואם if (dateIdx === -1 || closeIdx === -1) throw new Error("Missing Date or Close columns in Yahoo CSV"); // מעבר על השורות ואיכלוס ה-dateMap for (let j = 1; j < parsedCsv.length; j++) { const row = parsedCsv[j]; if (row.length <= Math.max(dateIdx, closeIdx)) continue; let dateStr = formatUniformDate(row[dateIdx]); let price = row[closeIdx]; // דילוג על שורות ריקות (ימי חג ביאהו מחזירים null) if (!dateStr || price === "null" || price === "") continue; // אתחול התאריך ב-Map אם טרם קיים if (!dateMap[dateStr]) dateMap[dateStr] = {}; dateMap[dateStr][ticker] = price; } } catch (yahooError) { // --- ניסיון גיבוי: Google Finance --- let tempSpreadsheetFile = null; try { const tempSpreadsheet = SpreadsheetApp.create(`TempDataFetch_${ticker}`); tempSpreadsheetFile = DriveApp.getFileById(tempSpreadsheet.getId()); const tempSheet = tempSpreadsheet.getSheets()[0]; // יצירת הנוסחה המיוחדת של גוגל פיננס const sParts = startDate.split("-"); const eParts = endDate.split("-"); const formula = `=GOOGLEFINANCE("${ticker}", "price", DATE(${sParts[0]},${sParts[1]},${sParts[2]}), DATE(${eParts[0]},${eParts[1]},${eParts[2]}), "${gfFreq}")`; tempSheet.getRange("A1").setFormula(formula); SpreadsheetApp.flush(); let gfData = null; // המתנה לאכלוס הנתונים (מניעת קריאת Loading) for (let retry = 0; retry < 6; retry++) { Utilities.sleep(2000); gfData = tempSheet.getDataRange().getValues(); if (gfData.length > 1 && gfData[0][0] !== "#N/A" && gfData[0][0] !== "Loading..." && gfData[0][0] !== "#ERROR!") break; } if (gfData && gfData.length > 1 && gfData[0][0] !== "#N/A" && gfData[0][0] !== "Loading...") { // דילוג על הכותרת (k=1) for (let k = 1; k < gfData.length; k++) { let dateStr = formatUniformDate(gfData[k][0]); let price = gfData[k][1]; if (!dateStr || price === "" || price === null) continue; if (!dateMap[dateStr]) dateMap[dateStr] = {}; dateMap[dateStr][ticker] = price; } } } catch (gfError) { // מניה שנכשלה בשני המקורות פשוט תישאר ריקה בטבלה } finally { // חובה להשמיד את הגיליון הזמני כדי למנוע הצפת חשבון הדרייב if (tempSpreadsheetFile) { tempSpreadsheetFile.setTrashed(true); } } } } // --- 4. בניית מערך הנתונים המשולב (Wide Format / Pivot Table) --- // חילוץ כל התאריכים הייחודיים שנאספו ומיון כרונולוגי const allDates = Object.keys(dateMap).sort((a, b) => new Date(a) - new Date(b)); // יצירת שורת הכותרות: "תאריך" ולאחריו כל סימולי המניות const combinedData = [ ["תאריך"].concat(tickersArray) ]; // בניית שורות הנתונים for (let i = 0; i < allDates.length; i++) { const currentDate = allDates[i]; const row = [currentDate]; // איסוף המחיר של כל מניה באותו תאריך (או תא ריק אם חסר) for (let j = 0; j < tickersArray.length; j++) { const tkr = tickersArray[j]; row.push(dateMap[currentDate][tkr] || ""); } combinedData.push(row); } let resultJSON = {}; // --- 5. טיפול בפעולה (Action) המבוקשת --- if (action === 'csv') { // המרת המערך הדו-ממדי למחרוזת CSV עם הגנה על תאים בעייתיים const csvString = combinedData.map(row => { return row.map(cell => { let str = String(cell); if (str.includes(',') || str.includes('"') || str.includes('\n')) { str = '"' + str.replace(/"/g, '""') + '"'; } return str; }).join(","); }).join("\n"); resultJSON = { "status": "success", "csvData": csvString }; } else if (action === 'sheet') { // יצירת גיליון תוצאה והזנת המערך const resultSpreadsheet = SpreadsheetApp.create("מחירי סגירה - נתונים היסטוריים"); const resultSheet = resultSpreadsheet.getSheets()[0]; if (combinedData.length > 0) { resultSheet.getRange(1, 1, combinedData.length, combinedData[0].length).setValues(combinedData); // עיצוב אסתטי לכותרת resultSheet.getRange(1, 1, 1, combinedData[0].length).setFontWeight("bold").setBackground("#f3f4f6"); } resultJSON = { "status": "success", "sheetUrl": resultSpreadsheet.getUrl() }; } else { throw new Error("Invalid action parameter. Must be 'csv' or 'sheet'."); } // --- 6. החזרת התשובה ללקוח --- return ContentService.createTextOutput(JSON.stringify(resultJSON)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { // בלוק השגיאות הראשי - מחזיר אובייקט JSON נקי כדי למנוע שגיאות CORS במסך הלקוח return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() })).setMimeType(ContentService.MimeType.JSON); } }
